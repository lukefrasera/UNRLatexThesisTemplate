\section{Update Loop}
\label{sec:updateloop}
\begin{algorithm}
\caption{Behavior update loop}
\label{alg:update}
\begin{algorithmic}[1]
\IF {$Not~Done$}
  \IF {$Is~Active$}
    \IF {$Preconditions$}
      \IF {$BEHAVIOR~NODE$}
      \STATE $Activate:$
        \STATE $~~~Mutex Acquired \Rightarrow state \leftarrow running$
      \ELSIF {$GOAL~NODE$}
        \STATE $Activate:$
          \STATE $~~~state \leftarrow done$
      \ENDIF
    \ELSE
      \STATE $SpreadActivation$
    \ENDIF
    \STATE \emph{ActivationFalloff} $\Rightarrow \alpha ~*$ \emph{activation\_level}  
  \ENDIF
\ENDIF
\end{algorithmic}
\label{update-loop}
\end{algorithm}
